import 'package:flutter/material.dart';
import 'dart:async';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        primarySwatch: Colors.deepOrange,
      ),
      home: MyHomePage(title: 'Widgets interactifs'),
      debugShowCheckedModeBanner: false,
    );
  }
}

class MyHomePage extends StatefulWidget {
  MyHomePage({Key key, this.title}) : super(key: key);

  final String title;

  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {

DateTime date;
TimeOfDay time;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.title),
        centerTitle: true,
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children:  [
            TextButton(
              child: Text(
                (date == null)? 'Appuyez-moi': date.toString()
              ),
              onPressed: monterDate, 
            ),
            TextButton(
              child: Text(
                (time == null)? 'Appuyez-moi': time.toString()
              ),
              onPressed: montrerHeure, 
            ),
          ],
        ),
      ),
    );
  }

// Fonctions ----------------------
  Future<Null> monterDate() async {
    DateTime choix = await showDatePicker(
      context: context, 
      initialDatePickerMode: DatePickerMode.year, // permet de choisir l'ann√©e avant le mois
      initialDate: DateTime.now(), 
      firstDate: DateTime(1983), 
      lastDate: DateTime(2045)
    );
    if (choix != null) {
      setState(() {
        date = choix;
      });
    }
  }

  Future<Null> montrerHeure() async {
    TimeOfDay heure = await showTimePicker(
      context: context, 
      initialTime: TimeOfDay.now()
    );
    if (heure != null) {
      setState(() {
        time = heure;
      });
    }
  }

}
